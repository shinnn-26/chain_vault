Smart Contract
On-chain access control & document ownership (Solidity)
ChainVault.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract ChainVault {
  struct Document {
    bytes32 fileHash;
    string ipfsCID;
    address owner;
    uint256 timestamp;
    bool isActive;
  }

  mapping(bytes32 => Document) public documents;
  mapping(address => bytes32[]) public userDocs;

  event DocumentStored(bytes32 indexed hash, address owner);
  event AccessGranted(bytes32 indexed hash, address user);

  function storeDocument(
    bytes32 _hash, string _cid
  ) external {
    documents[_hash] = Document(_hash, _cid, msg.sender, block.timestamp, true);
    emit DocumentStored(_hash, msg.sender);
  }

  function verifyDocument(
    bytes32 _hash
  ) external view returns (bool) {
    return documents[_hash].isActive;
  }
}
